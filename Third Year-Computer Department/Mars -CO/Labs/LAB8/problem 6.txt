.data
newline: .asciiz "\n"
prompt_x: .asciiz "Enter X : "
appro: .asciiz "squareRoot approximately is: "
actually: .asciiz "squareRoot actually is: "
err: .asciiz "error is: "
num1: .double 1.00
num2: .double 2.00

.text

.globl main

main:

    # Prompt the user to enter the coefficients
    li $v0, 4
    la $a0, prompt_x
    syscall

    # Read coefficient a from the user
    li $v0, 7
    syscall
    mov.d $f20, $f0
    
    jal squareRoot_appr
    
    jal squareRoot
    
    jal error 
   
    # Exit program
    li $v0, 10
    syscall
   
    
squareRoot_appr:

   li $t0,0
   ldc1 $f2,num1
   ldc1 $f4,num2
   
loop:

  
   div.d $f6,$f20,$f2
   add.d $f6,$f6,$f2
   div.d $f2,$f6,$f4
   addi $t0,$t0,1
   blt $t0,20,loop 
   
    # Display the roots
    li $v0, 4
    la $a0,appro
    syscall

    li $v0, 3
    mov.d $f12, $f2
    syscall

    # Print a newline character
    la $a0, newline            # Load the address of the newline string
    li $v0, 4                  # System call code for printing a string
    syscall
   
   jr $ra

squareRoot:

   sqrt.d $f0,$f20

    li $v0, 4
    la $a0, actually
    syscall


    li $v0, 3
    mov.d $f12, $f0
    syscall

    # Print a newline character
    la $a0, newline            # Load the address of the newline string
    li $v0, 4                  # System call code for printing a string
    syscall
   
   jr $ra

error:
   
   sub.d $f8,$f0,$f2
   
    li $v0, 4
    la $a0, err
    syscall
   
    li $v0, 3
    mov.d $f12, $f8
    syscall

    # Print a newline character
    la $a0, newline            # Load the address of the newline string
    li $v0, 4                  # System call code for printing a string
    syscall
   
   jr $ra
